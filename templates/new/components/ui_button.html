{{define "ui_button"}} {{- $variant := default "primary" .variant -}} {{- $size := default "md" .size -}} {{- $btnClass := "" -}} {{- if eq $variant "primary" -}}{{- $btnClass = "btn-primary" -}} {{- else if eq $variant "secondary" -}}{{- $btnClass = "btn-secondary" -}} {{- else if eq $variant "danger" -}}{{- $btnClass = "btn-danger" -}} {{- else if eq $variant "ghost" -}}{{- $btnClass = "btn-ghost" -}} {{- else if eq $variant "icon-only" -}}{{- $btnClass = "btn-icon-only" -}} {{- end -}} {{- if eq $size "sm" -}}{{- $btnClass = print $btnClass " btn-sm" -}}{{- end -}} {{end}}
