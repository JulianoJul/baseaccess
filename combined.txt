        // ================================================================

        let db = null;
        let SQL = null;
        let expedientesCache = [];
        let catalogosCache = {};
        let dbFilePath = null;
        let dbModificado = false;
        let autosaveTimer = null;
        // Campos identificados como "de edición constante" (histórico de Tablas7).
        // No se derivan del trigger de Tablas8 porque ese trigger amplió el snapshot
        // por completitud, no porque esos campos se editen con más frecuencia.
        const CAMPOS_EDICION_FRECUENTE = [
            'id_tipo_contrato', 'id_emisor', 'id_receptor', 'id_gerencia',
            'id_superintendencia', 'id_documento', 'id_estatus',
            'fecha_recibido', 'fecha_devuelto', 'observaciones_generales'
        ];
        let snapshotEdicion = null;
        let observacionesPreviasCache = '';

        function marcarCamposEdicionFrecuente() {
            CAMPOS_EDICION_FRECUENTE.forEach(campo => {
                const el = document.getElementById(`f-${campo}`);
                if (!el) return;
                const contenedor = el.closest('div');
                if (contenedor && !contenedor.classList.contains('campo-frecuente')) {
                    contenedor.classList.add('campo-frecuente');
                    const label = contenedor.querySelector('.label');
                    if (label) {
                        label.title = 'Este campo cambia con frecuencia en el flujo del expediente';
                    }
                }
            });
        }

        function obtenerTextoSelect(id) {
            const el = document.getElementById(id);
            if (!el || !el.options) return '';
            return el.options[el.selectedIndex] ? el.options[el.selectedIndex].text : '';
        }

        function toggleObsPrevias() {
            const wrapper = document.getElementById('obs-previas-wrapper');
            const contenido = document.getElementById('obs-previas-contenido');
            if (!wrapper || !contenido) return;
            const btn = wrapper.querySelector('button');
            if (contenido.classList.contains('hidden')) {
                contenido.classList.remove('hidden');
                if (btn) btn.innerHTML = '<i class="fas fa-eye-slash mr-1"></i>Ocultar observaciones anteriores';
            } else {
                contenido.classList.add('hidden');
                if (btn) btn.innerHTML = '<i class="fas fa-eye mr-1"></i>Ver observaciones anteriores';
            }
        }

        function generarObservacionAutomatica(textoLibre) {
            const ahora = new Date().toLocaleDateString('es-VE', { year: 'numeric', month: '2-digit', day: '2-digit' });
            if (!snapshotEdicion) {
                const estatus = obtenerTextoSelect('f-id_estatus') || 'PENDIENTE';
                const doc = obtenerTextoSelect('f-id_documento') || '';
                const recibido = document.getElementById('f-fecha_recibido')?.value || '';
                const devuelto = document.getElementById('f-fecha_devuelto')?.value || '';
                let msg = `[${ahora}] Expediente creado — Estatus: ${estatus}`;
                if (doc) msg += ` — Documento: ${doc}`;
                if (recibido) msg += ` — F. Recibido: ${recibido}`;
                if (devuelto) msg += ` — F. Devuelto: ${devuelto}`;
                if (textoLibre) msg += ` — ${textoLibre}`;
                return msg;
            }
            const cambios = [];
            CAMPOS_EDICION_FRECUENTE.forEach(campo => {
                if (campo === 'observaciones_generales') return;
                const el = document.getElementById(`f-${campo}`);
                if (!el) return;
                const valorActual = el.value || '';
                const valorAnterior = (snapshotEdicion[campo] || '');
                if (String(valorActual) !== String(valorAnterior)) {
                    const textoActual = (el.tagName === 'SELECT') ? (el.options[el.selectedIndex]?.text || valorActual) : valorActual;
                    const textoAnterior = snapshotEdicion[`_txt_${campo}`] || valorAnterior;
                    cambios.push(`${campo.replace(/_/g,' ')}: ${textoAnterior || '(vacio)'} → ${textoActual || '(vacio)'}`);
                }
            });
            if (cambios.length === 0 && !textoLibre) return null;
            let msg = `[${ahora}] ${cambios.join(' | ')}`;
            if (textoLibre) msg += ` — ${textoLibre}`;
            return msg;
        }

        const CATALOGO_POR_SELECT = {
            'f-id_gerencia':         { key: 'gerencia',           tabla: 'cat_gerencia',         cols: 'id, nombre' },
            'f-id_superintendencia': { key: 'superintendencia',   tabla: 'cat_superintendencia', cols: 'id, nombre, id_gerencia', extra: { col: 'id_gerencia', label: 'ID Gerencia', selectId: 'f-id_gerencia' } },
            'f-id_documento':        { key: 'documento',          tabla: 'cat_documento',        cols: 'id, nombre' },
            'f-id_plan':             { key: 'plan_contratacion',  tabla: 'cat_plan_contratacion', cols: 'id, nombre' },
            'f-id_modalidad':        { key: 'modalidad',          tabla: 'cat_modalidad',        cols: 'id, nombre' },
            'f-id_art':              { key: 'art',                tabla: 'cat_art',              cols: 'id, nombre' },
            'f-id_tipo_contrato':    { key: 'tipo_contrato',      tabla: 'cat_tipo_contrato',    cols: 'id, nombre' },
            'f-id_estatus':          { key: 'estatus_detalle',    tabla: 'cat_estatus_detalle',  cols: 'id, nombre' },
            'f-id_resultado':        { key: 'resultado_proceso',  tabla: 'cat_resultado_proceso', cols: 'id, nombre' },
            'f-id_empresa':          { key: 'empresas',           tabla: 'cat_empresas',         cols: 'id, nombre' },

            'f-id_receptor':         { key: 'responsables',       tabla: 'cat_responsables',     cols: 'id, nombre' }

        function inicializarBotonesCatalogo() {
            const form = document.getElementById('form-modal');
            if (!form) return;
            form.querySelectorAll('[id^="f-"]').forEach(el => {
                if (el.tagName !== 'SELECT') return;
                if (!CATALOGO_POR_SELECT[el.id]) return;
                const label = el.closest('div')?.querySelector('.label');
                if (!label || label.querySelector('.btn-catalogo')) return;
                const btn = document.createElement('button');
                btn.type = 'button';
                btn.className = 'btn-catalogo text-[10px] text-gray-500 hover:text-teal-400 ml-1';
                btn.title = 'Agregar nuevo registro al catálogo';
                btn.innerHTML = '<i class="fas fa-plus"></i>';
                btn.onclick = (e) => { e.stopPropagation(); abrirAgregarCatalogo(el.id); };
                label.appendChild(btn);
            });
        }

        function abrirAgregarCatalogo(selectId) {
            const info = CATALOGO_POR_SELECT[selectId];
            if (!info) return;
            const modal = document.getElementById('modal-agregar-catalogo');
            if (!modal) return;

        function guardarNuevoCatalogo() {
            const selectId = document.getElementById('ac-select-id').value;
            const catKey = document.getElementById('ac-catalog-key').value;
            const tabla = document.getElementById('ac-tabla').value;
            const nombre = document.getElementById('ac-nombre').value.trim();
            if (!nombre) { alert('El nombre es obligatorio.'); document.getElementById('ac-nombre').focus(); return; }

            const info = CATALOGO_POR_SELECT[selectId];
            let cols = 'nombre';
            let values = "'" + nombre.replace(/'/g, "''") + "'";
            let placeholders = ['nombre'];

            if (info && info.extra) {
                let extraVal;
                if (info.extra.selectId) {
                    const sel = document.getElementById(info.extra.selectId);
                    extraVal = sel ? sel.value : null;
                } else {
                    extraVal = document.getElementById('ac-extra-val').value.trim() || info.extra.def || null;
                }
                if (extraVal) {
                    cols += ', ' + info.extra.col;
                    values += ', ' + extraVal;
                    placeholders.push(info.extra.col);
                }
            }

            try {
                const sql = `INSERT INTO ${tabla} (${cols}) VALUES (${values})`;
                db.exec(sql);
                dbModificado = true;
                // Recargar el catálogo y repoblar
                const catalogInfo = { id: catKey, tabla, cols: 'id, nombre' };
                if (info && info.extra) catalogInfo.cols += ', ' + info.extra.col;
                const res = db.exec(`SELECT ${catalogInfo.cols} FROM ${tabla} ORDER BY nombre`);
                catalogosCache[catKey] = dbToObjects(res);
                poblarSelectores();
                // Seleccionar el nuevo registro buscándolo por nombre
                const nuevoItem = (catalogosCache[catKey] || []).find(item => item.nombre === nombre);
                if (nuevoItem && document.getElementById(selectId)) {
                    document.getElementById(selectId).value = nuevoItem.id;
                }
                cerrarAgregarCatalogo();
            } catch (err) {
                alert('Error al guardar: ' + err.message);
            }
        }

        function _cargarBaseDatosComun(bytes, fileName, filePath) {
            const signature = new TextDecoder().decode(bytes.slice(0, 16));
            if (!signature.startsWith('SQLite format 3')) {
                alert('El archivo no parece ser una base de datos SQLite válida.');
                return false;
            }
            db = new SQL.Database(bytes);
            dbFilePath = filePath || fileName;
            if (electronAPI && filePath) electronAPI.setDbPath(filePath);
            document.getElementById('search').disabled = false;
            document.getElementById('btn-nuevo').disabled = false;
            document.getElementById('btn-guardar').disabled = false;
            document.getElementById('dbfile').value = '';
            document.getElementById('body').classList.remove('drag-over');
            actualizarEstadoBD(`<i class="fas fa-check"></i> ${fileName}`);
            cargarCatalogos();
            cargarDatos();
            iniciarAutoguardado();
            registrarReciente(fileName, filePath);
            return true;
        }

        function registrarReciente(nombre, path) {
            if (!nombre) return;
            let recientes = [];
            try {
                recientes = JSON.parse(localStorage.getItem('recientes') || '[]');
            } catch(e) { recientes = []; }
            const idx = recientes.findIndex(r => r.path === path);
            if (idx !== -1) recientes.splice(idx, 1);
            recientes.unshift({ nombre, path, timestamp: Date.now() });
            if (recientes.length > 8) recientes = recientes.slice(0, 8);
            localStorage.setItem('recientes', JSON.stringify(recientes));
        }

        function eliminarReciente(path) {
            let recientes = [];
            try { recientes = JSON.parse(localStorage.getItem('recientes') || '[]'); } catch(e) {}
            recientes = recientes.filter(r => r.path !== path);
            localStorage.setItem('recientes', JSON.stringify(recientes));
            mostrarMenuRecientes();
        }

        function abrirBaseDatosReciente(path) {
            if (!path) return;
            if (electronAPI && electronAPI.openDbFilePath) {
                electronAPI.openDbFilePath(path).then(dataBase64 => {
                    if (!dataBase64) return;
                    const bytes = Uint8Array.from(atob(dataBase64), c => c.charCodeAt(0));
                    const name = path.split(/[\\/]/).pop();
                    _cargarBaseDatosComun(bytes, name, path);
                }).catch(err => {
                    alert('No se pudo abrir la base de datos: ' + err.message);
                    eliminarReciente(path);
                });
            } else {
                const input = document.getElementById('dbfile');
                if (input) input.click();
            }
            document.getElementById('menu-recientes')?.classList.add('hidden');
        }

        async function abrirBaseDatos() {
            console.log('[RENDERER] Función abrirBaseDatos llamada');
            if (electronAPI && electronAPI.openDbDialog) {
                console.log('[RENDERER] electronAPI disponible, intentando abrir diálogo...');

        async function guardarBD() {
            if (!db) return;
            try {
                const data = db.export();
                let binary = '';
                const chunkSize = 8192;
                for (let i = 0; i < data.length; i += chunkSize) {
                    binary += String.fromCharCode.apply(null, data.subarray(i, Math.min(i + chunkSize, data.length)));
                }
                const data_base64 = btoa(binary);
                if (electronAPI && dbFilePath) {
                    await electronAPI.saveDb(data_base64);
                }
                dbModificado = false;
                actualizarEstadoBD('<i class="fas fa-check"></i> Guardado');
            } catch (err) {
                console.error('Error al guardar BD:', err);
            }
        }

        document.addEventListener('drop', async (e) => {
            e.preventDefault();
            document.getElementById('body').classList.remove('drag-over');

            if (!file) return;
            if (!file.name.toLowerCase().match(/\.(db|sqlite)$/)) {
                alert('Solo se aceptan archivos .db o .sqlite');
                return;
            }
            const MAX_SIZE = 100 * 1024 * 1024;
            if (file.size > MAX_SIZE) {
                alert(`El archivo es demasiado grande (${(file.size / 1024 / 1024).toFixed(2)} MB). Máximo: 100 MB`);
                return;
            }
            try {
                const reader = new FileReader();
                const buffer = await new Promise((resolve, reject) => {
                    reader.onload = () => resolve(reader.result);
                    reader.onerror = reject;
                    reader.readAsArrayBuffer(file);
                });
                const bytes = new Uint8Array(buffer);
                _cargarBaseDatosComun(bytes, file.name, file.path || file.name);
            } catch (err) {
                console.error('Error en drop:', err);
                alert('Error al abrir archivo: ' + err.message);
            }
        });

        function execSafe(sql, params) {
            if (!db) { alert('Primero carga un archivo .db'); return null; }
            try {
                if (params && params.length) {
                    const escaped = sql.replace(/\?/g, () => {
                        const v = params.shift();
                        if (v === null || v === undefined) return 'NULL';
                        if (typeof v === 'number') return String(v);
                        return "'" + String(v).replace(/'/g, "''") + "'";
                    });
                    return db.exec(escaped);
                }
                return db.exec(sql);
            } catch (err) {
                alert('Error en consulta: ' + err.message);
                return null;
            }
        }

        function toInt(v) { const n = parseInt(v, 10); return isNaN(n) ? null : n; }

        function dbToObjects(res) {
            if (!res || res.length === 0) return [];
            const columns = res[0].columns;
            return res[0].values.map(row => {
                let obj = {};
                columns.forEach((col, i) => obj[col] = row[i]);
                return obj;
            });
        }

        function sanitizeNull(val) {
            if (val === null || val === undefined || val === '') return null;
            return val;
        }

        function cargarCatalogos() {
            const vistos = new Set();
            Object.values(CATALOGO_POR_SELECT).forEach(info => {

                catalogosCache[info.key] = dbToObjects(res);

            poblarSelectores();

        function formatNum(v) {
            if (v === '' || v === null || v === undefined) return '';
            if (typeof v === 'number') {
                return v.toLocaleString('es-ES', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
            }
            let s = String(v).replace(/\./g, '').replace(',', '.');
            let n = parseFloat(s);
            if (isNaN(n)) return String(v);
            return n.toLocaleString('es-ES', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
        }
        function parseNum(v) {
            if (v === '' || v === null || v === undefined) return '';
            return String(v).replace(/\./g, '').replace(',', '.');
        }

        function limitDecimals(v) {
            let s = String(v).split(',');
            if (s.length > 2) {
                s = s[0] + ',' + s.slice(1).join('');
            } else {
                s = v;
            }
            let parts = String(s).split(',');
            if (parts.length === 2 && parts[1].length > 2) {
                return parts[0] + ',' + parts[1].slice(0, 2);
            }
            return s;
        }

        function validarFechas() {
            const recibido = document.getElementById('f-fecha_recibido').value;
            const devuelto = document.getElementById('f-fecha_devuelto').value;
            const errEl = document.getElementById('f-error-fechas');
            if (recibido && devuelto && devuelto < recibido) {
                if (!errEl) {
                    const div = document.createElement('div');
                    div.id = 'f-error-fechas';
                    div.className = 'text-red-400 text-xs mt-1';
                    div.innerHTML = '<i class="fas fa-exclamation-triangle text-red-400 mr-1"></i> Fecha Devuelto no puede ser anterior a Fecha Recibido';
                    document.getElementById('f-fecha_devuelto').parentNode.appendChild(div);
                }
                return false;
            }
            if (errEl) errEl.remove();
            return true;
        }

        function _queryHistorial(safeId) {
            const r = db.exec(`
                SELECT h.id_movimiento, COALESCE(g.nombre,'N/A') AS ger,
                    COALESCE(em.nombre,'N/A') AS emisor, COALESCE(rec.nombre,'N/A') AS receptor,
                    COALESCE(ed.nombre,'N/A') AS estatus, COALESCE(h.fecha_recibido,'') AS f_entrada,
                    COALESCE(h.fecha_devuelto,'') AS f_salida, h.observaciones_generales
                FROM historial_movimientos h
                LEFT JOIN cat_gerencia g ON h.id_gerencia = g.id
                LEFT JOIN cat_responsables em ON h.id_emisor = em.id
                LEFT JOIN cat_responsables rec ON h.id_receptor = rec.id
                LEFT JOIN cat_estatus_detalle ed ON h.id_estatus = ed.id
                WHERE h.id_expediente = ${safeId}
                ORDER BY h.id_movimiento DESC
            `);
            return dbToObjects(r);
        }

CREATE TRIGGER trg_exp_auditoria AFTER UPDATE ON expedientes
FOR EACH ROW
WHEN (
    OLD.id_tipo_contrato IS NOT NEW.id_tipo_contrato OR
    OLD.id_emisor IS NOT NEW.id_emisor OR
    OLD.id_receptor IS NOT NEW.id_receptor OR
    OLD.id_gerencia IS NOT NEW.id_gerencia OR
    OLD.id_superintendencia IS NOT NEW.id_superintendencia OR
    OLD.id_documento IS NOT NEW.id_documento OR
    OLD.id_estatus IS NOT NEW.id_estatus OR
    OLD.fecha_recibido IS NOT NEW.fecha_recibido OR
    OLD.fecha_devuelto IS NOT NEW.fecha_devuelto OR
    OLD.nro_proceso IS NOT NEW.nro_proceso OR
    OLD.presupuesto_base_usd IS NOT NEW.presupuesto_base_usd OR
    OLD.tipo_cambio IS NOT NEW.tipo_cambio OR
    OLD.monto_adjudicado_usd IS NOT NEW.monto_adjudicado_usd OR
    OLD.id_resultado IS NOT NEW.id_resultado OR
    OLD.id_empresa IS NOT NEW.id_empresa OR
    OLD.tiempo_ejecucion IS NOT NEW.tiempo_ejecucion OR
    OLD.fecha_firma_contrato IS NOT NEW.fecha_firma_contrato OR
    OLD.observaciones_generales IS NOT NEW.observaciones_generales
)
BEGIN
    INSERT INTO historial_movimientos (
        id_expediente, id_tipo_contrato, id_gerencia, id_superintendencia,
        id_documento, id_emisor, id_receptor, id_estatus,
        fecha_recibido, fecha_devuelto, nro_proceso,
        presupuesto_base_usd, tipo_cambio, monto_adjudicado_usd,
        id_resultado, id_empresa, tiempo_ejecucion, fecha_firma_contrato,
        observaciones_generales
    ) VALUES (
        NEW.id_expediente, NEW.id_tipo_contrato, NEW.id_gerencia, NEW.id_superintendencia,
        NEW.id_documento, NEW.id_emisor, NEW.id_receptor, NEW.id_estatus,
        NEW.fecha_recibido, NEW.fecha_devuelto, NEW.nro_proceso,
        NEW.presupuesto_base_usd, NEW.tipo_cambio, NEW.monto_adjudicado_usd,
        NEW.id_resultado, NEW.id_empresa, NEW.tiempo_ejecucion, NEW.fecha_firma_contrato,
        NEW.observaciones_generales
    );

    UPDATE expedientes
    SET id_estatus = (SELECT id FROM cat_estatus_detalle WHERE nombre = 'PENDIENTE' LIMIT 1)
    WHERE NEW.fecha_firma_contrato IS NULL
      AND OLD.fecha_firma_contrato IS NOT NULL
      AND id_expediente = NEW.id_expediente;

    UPDATE expedientes
    SET fecha_actualizacion = CURRENT_DATE
    WHERE id_expediente = NEW.id_expediente;
END;

const { app, BrowserWindow, ipcMain, dialog } = require('electron')
const path = require('path')
const fs = require('fs')

console.log('[MAIN] Iniciando proceso principal...')

let currentDbPath = null

function createWindow() {
  const win = new BrowserWindow({
    width: 1400,
    height: 900,
    title: 'Gestión de Expedientes con Historial',
    webPreferences: {
      preload: path.join(__dirname, 'preload.js'),
      nodeIntegration: false,
      contextIsolation: true,
    }
  })

  win.loadFile('index.html')
  win.setMenuBarVisibility(false)

  win.webContents.on('before-input-event', (_event, input) => {
    if (input.key === 'F12') {
      win.webContents.toggleDevTools()
    }
  })
}

// IPC handlers
ipcMain.handle('save-db', async (_event, dataBase64) => {
  if (!currentDbPath) throw new Error('No hay archivo abierto')
  const data = Buffer.from(dataBase64, 'base64')
  fs.writeFileSync(currentDbPath, data)
  return true
})

ipcMain.handle('save-db-as', async (event, dataBase64) => {
  const win = BrowserWindow.fromWebContents(event.sender)
  if (!win) return null
  const result = await dialog.showSaveDialog(win, {
    filters: [{ name: 'SQLite DB', extensions: ['db', 'sqlite'] }],
  })
  if (result.canceled || !result.filePath) {
    return null
  }
  const data = Buffer.from(dataBase64, 'base64')
  fs.writeFileSync(result.filePath, data)
  currentDbPath = result.filePath
  return result.filePath
})

ipcMain.handle('set-db-path', async (_event, filePath) => {
  currentDbPath = filePath
  return true
})

ipcMain.handle('get-db-path', async () => {
  return currentDbPath
})

ipcMain.handle('open-db-file', async (_event, filePath) => {
  currentDbPath = filePath
  const data = fs.readFileSync(filePath)
  return data.toString('base64')
})

ipcMain.handle('open-db-dialog', async (event) => {
  console.log('[MAIN] Recibida solicitud open-db-dialog')
  const win = BrowserWindow.fromWebContents(event.sender)
  if (!win) {
    console.error('[MAIN] No se pudo obtener BrowserWindow desde el sender')
    return null
  }
  console.log('[MAIN] Ventana encontrada, abriendo diálogo...')
  const result = await dialog.showOpenDialog(win, {
    filters: [{ name: 'SQLite DB', extensions: ['db', 'sqlite'] }],
    properties: ['openFile'],
    title: 'Seleccionar Base de Datos',
  })
  console.log('[MAIN] Resultado del diálogo:', result)
  if (result.canceled || result.filePaths.length === 0) return null
  const filePath = result.filePaths[0]
  console.log('[MAIN] Archivo seleccionado:', filePath)
  currentDbPath = filePath
  const data = fs.readFileSync(filePath)
  return { path: filePath, data: data.toString('base64') }
})

app.whenReady().then(createWindow)
app.on('window-all-closed', () => app.quit())

## Contexto Termux (Android)

Este proyecto se edita y construye desde **Termux** en Android. Si inicias una sesión nueva:

El build se genera en `dist/win-unpacked/`. Copiar esa carpeta a USB y ejecutar `GestionExpedientes.exe`.

**Importante:** `node_modules/` y `dist/` no se suben a git (`.gitignore`). Hay que reinstalar dependencias cada sesión nueva.

## ⚠️ Limitación: `file://` + WASM

Al abrir `index.html` con doble click (`file://` protocol), los navegadores **bloquean la carga del binario WASM** por seguridad. Síntomas:
- El botón "+ Nuevo Expediente" queda deshabilitado
- Los registros de la BDD no se muestran en la tabla

**Usar siempre Electron WinUnpacked** (`dist/win-unpacked/GestionExpedientes.exe`) para evitar este problema.

## Arquitectura

App web 100% cliente-side. **HTML + Tailwind CSS = UI** | **sql.js (SQLite WASM) = Data Layer**.
Sin backend, sin servidor, sin runtime externo. Un solo archivo HTML.

Dos modos de ejecución:

1. **Navegador** — abrir `index.html` directo (dependencias locales en `vendor/`)
2. **Electron WinUnpacked** — `GestionExpedientes.exe` con Chromium embebido (sin depender de Firefox/Chrome)

## Reglas del Proceso

1. **doc.md primero**: antes de cualquier implementación o cambio de código, actualizar esta documentación con lo que se planea hacer.
2. **Makefile siempre**: después de cambios, ejecutar `make combine`.
3. **Sin hardcodeo**: cero assumptions de naming conventions. Toda heurística debe ser configurable.
4. **Historial de cambios**: cada cambio debe agregarse a la cronología en `doc.md` con fecha, archivo, y razón.
5. **DRY + Reutilización**: toda pieza de lógica debe tener una representación única. No repetir código ni copiar-pegar bloques. Si un patrón aparece en más de un lugar, extraer a función reutilizable. La modularidad no se mide en líneas por archivo ni por función, sino en ausencia de redundancia y en que cada función tenga una única responsabilidad (SRP). Una función de 200 líneas sin duplicación interna es mejor que 4 funciones de 50 líneas con lógica repetida.

### Bug de persistencia resuelto (Electron)

Antes: sql.js modificaba la BD en RAM, nunca escribía al disco.
Ahora: se agregó `preload.js` + IPC handlers en `main.js` para leer/escribir archivos `.db`. Después de cada `guardarExpediente()` y `eliminarExpediente()`, se exporta el buffer de sql.js (`db.export()`) y se escribe al archivo `.db` vía `electronAPI.saveDb()`. Además hay autoguardado cada 30s, al cerrar la ventana, y atajo Ctrl+S.

### Apertura de Base de Datos (Electron)

El flujo de apertura usa **`<input type="file">` nativo del navegador** (no IPC), por confiabilidad:
1. El botón "Abrir Base de Datos" dispara un `<input type="file" id="dbfile" accept=".db,.sqlite" class="hidden">`.
2. El `change` event lee el archivo con `FileReader` → `Uint8Array` → `new SQL.Database(bytes)`.
3. La ruta del archivo se obtiene de `f.path` (propiedad nativa de Electron/Chromium para drag & drop y file input).
4. Se sincroniza con el backend vía `electronAPI.setDbPath(f.path)` para que `saveDb()` sepa dónde escribir.
5. Drag & drop: mismo flujo via `FileReader` + `file.path`.

**Por qué no IPC para abrir:** El `<input type="file">` es un estándar web que funciona siempre, sin depender de preload/contextBridge. En la primera versión se intentó con IPC (`pickDbFile` → `dialog.showOpenDialog`) pero fallaba en ciertos entornos (Windows sin focus, problemas con `getWindow()`).

### Rama `tauri-migration`

Existe la rama `tauri-migration` que reemplaza Electron por Tauri v2 (Rust). `master` queda intacto con Electron. Ver esa rama para los detalles de la migración.
