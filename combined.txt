    <script>
        // ================================================================
        // ESTADO GLOBAL
        // ================================================================
        let db = null;
        let expedientesCache = [];
        let catalogosCache = {};

        const config = { locateFile: filename => `vendor/${filename}` };
        
        initSqlJs(config).then(SQL => {
            document.getElementById('dbfile').addEventListener('change', function(e) {
                const f = e.target.files[0];
                if (!f) return;
                const r = new FileReader();
                r.onload = function() {
                    const Uints = new Uint8Array(r.result);
                    db = new SQL.Database(Uints);
                    document.getElementById('search').disabled = false;
                    document.getElementById('btn-nuevo').disabled = false;
                    cargarCatalogos();
                    cargarDatos();
                }
                r.readAsArrayBuffer(f);
            });
        });

        // ================================================================
        // UTILIDADES BD
        // ================================================================
        function dbToObjects(res) {
            if (!res || res.length === 0) return [];
            const columns = res[0].columns;
            return res[0].values.map(row => {
                let obj = {};
                columns.forEach((col, i) => obj[col] = row[i]);
                return obj;
            });
        }

        function escapeSql(val) {
            if (val === null || val === undefined || val === '') return null;
            return val;
        }

        // ================================================================
        // CATÁLOGOS
        // ================================================================
        function cargarCatalogos() {
            const tablas = [
                { id: 'gerencia',           tabla: 'cat_gerencia', cols: 'id, nombre' },
                { id: 'superintendencia',   tabla: 'cat_superintendencia', cols: 'id, nombre, id_gerencia' },
                { id: 'documento',          tabla: 'cat_documento', cols: 'id, nombre' },
                { id: 'plan_contratacion',  tabla: 'cat_plan_contratacion', cols: 'id, nombre' },
                { id: 'modalidad',          tabla: 'cat_modalidad', cols: 'id, nombre' },
                { id: 'art',                tabla: 'cat_art', cols: 'id, nombre' },
                { id: 'tipo_contrato',      tabla: 'cat_tipo_contrato', cols: 'id, nombre' },
                { id: 'estatus_detalle',    tabla: 'cat_estatus_detalle', cols: 'id, nombre' },
                { id: 'resultado_proceso',  tabla: 'cat_resultado_proceso', cols: 'id, nombre' },
                { id: 'empresas',           tabla: 'cat_empresas', cols: 'id, nombre' },
                { id: 'responsables',       tabla: 'cat_responsables', cols: 'id, nombre' },
                { id: 'estado_accion',      tabla: 'cat_estado_accion', cols: 'id, nombre' }
            ];
            tablas.forEach(t => {
                const res = db.exec(`SELECT ${t.cols} FROM ${t.tabla} ORDER BY nombre`);
                catalogosCache[t.id] = dbToObjects(res);
            });
            poblarSelectores();
        }

        function poblarSelectores() {
            const mapeo = {
                'f-id_gerencia':         'gerencia',
                'f-id_superintendencia': 'superintendencia',
                'f-id_documento':        'documento',
                'f-id_plan':             'plan_contratacion',
                'f-id_modalidad':        'modalidad',
                'f-id_art':              'art',
                'f-id_tipo_contrato':    'tipo_contrato',
                'f-id_estatus':          'estatus_detalle',
                'f-id_resultado':        'resultado_proceso',
                'f-id_empresa':          'empresas',
                'f-id_emisor':           'responsables',
                'f-id_receptor':         'responsables',
                'f-id_estado_accion':    'estado_accion'
            };
            Object.entries(mapeo).forEach(([selectId, catKey]) => {
                const sel = document.getElementById(selectId);
                if (!sel) return;
                sel.innerHTML = '<option value="">-- Seleccionar --</option>';
                catalogosCache[catKey].forEach(item => {
                    const opt = document.createElement('option');
                    opt.value = item.id;
                    opt.textContent = item.nombre;
                    sel.appendChild(opt);
                });
            });
            // Cargar superintendencias según gerencia inicial
            cargarSuperintendencias();
        }

        function cargarSuperintendencias() {
            const gerenciaId = document.getElementById('f-id_gerencia').value;
            const sel = document.getElementById('f-id_superintendencia');
            sel.innerHTML = '<option value="">-- Seleccionar --</option>';
            catalogosCache['superintendencia']
                .filter(item => !gerenciaId || item.id_gerencia === parseInt(gerenciaId))
                .forEach(item => {
                    const opt = document.createElement('option');
                    opt.value = item.id;
                    opt.textContent = item.nombre;
                    sel.appendChild(opt);
                });
        }

        // ================================================================
        // CARGA Y RENDERIZADO DE TABLA
        // ================================================================
        function cargarDatos() {
            if (!db) return;
            const res = db.exec("SELECT * FROM vw_reporte_excel_contrataciones ORDER BY id_expediente DESC");
            expedientesCache = dbToObjects(res);
            renderizarTabla(expedientesCache);
        }

        function renderizarTabla(lista) {
            const cuerpo = document.getElementById('tabla-cuerpo');
            cuerpo.innerHTML = '';

            if (lista.length === 0) {
                cuerpo.innerHTML = `<tr><td colspan="8" class="p-8 text-center text-gray-500">No se encontraron expedientes con esos criterios.</td></tr>`;
                return;
            }

            lista.forEach(exp => {
                const filaPrincipal = document.createElement('tr');
                filaPrincipal.className = "hover:bg-gray-700/20 transition-colors cursor-pointer border-gray-700";
                filaPrincipal.innerHTML = `
                    <td class="p-3 text-center">
                        <button onclick="toggleDesplegable(${exp.id_expediente}, event)" class="text-teal-400 font-bold hover:text-teal-300 text-base focus:outline-none" id="btn-${exp.id_expediente}">➕</button>
                    </td>
                    <td class="p-3 text-center">
                        <button onclick="event.stopPropagation(); mostrarFormulario(${exp.id_expediente})" class="text-blue-400 hover:text-blue-300 text-sm focus:outline-none" title="Editar expediente">✏️</button>
                    </td>
                    <td class="p-3 font-semibold text-gray-200">${exp.solped || 'SIN SOLPED'}</td>
                    <td class="p-3 text-gray-300">${exp.gerencia || 'N/A'}</td>
                    <td class="p-3 text-gray-400 max-w-xs truncate" title="${exp.documento}">${exp.documento || 'N/A'}</td>
                    <td class="p-3">
                        <span class="px-2 py-0.5 rounded-full text-[11px] font-medium ${getEstatusClass(exp.estatus_detalle)}">
                            ${exp.estatus_detalle || 'PENDIENTE'}

        function toggleDesplegable(id, event) {
            if(event) event.stopPropagation();
            const subfila = document.getElementById(`subfila-${id}`);
            const btn = document.getElementById(`btn-${id}`);
            if (subfila.classList.contains('hidden')) {
                subfila.classList.remove('hidden');
                btn.innerText = '➖';
                cargarUltimoMovimiento(id);
            } else {
                subfila.classList.add('hidden');
                btn.innerText = '➕';
            }
        }

        function cargarUltimoMovimiento(id) {
            const contenedor = document.getElementById(`hist-content-${id}`);
            const res = db.exec(`
                SELECT h.id_movimiento,
                    COALESCE(g.nombre,'N/A') AS ger, COALESCE(em.nombre,'N/A') AS emisor,
                    COALESCE(rec.nombre,'N/A') AS receptor, COALESCE(ed.nombre,'N/A') AS estatus,
                    COALESCE(h.fecha_recibido,'') AS f_entrada, COALESCE(h.fecha_devuelto,'') AS f_salida
                FROM historial_movimientos h
                LEFT JOIN cat_gerencia g ON h.id_gerencia = g.id
                LEFT JOIN cat_responsables em ON h.id_emisor = em.id
                LEFT JOIN cat_responsables rec ON h.id_receptor = rec.id
                LEFT JOIN cat_estatus_detalle ed ON h.id_estatus = ed.id
                WHERE h.id_expediente = ${id}
                ORDER BY h.id_movimiento DESC LIMIT 1
            `);
            const data = dbToObjects(res);
            if (data.length === 0) {
                contenedor.innerHTML = '<p class="text-gray-500 italic p-2">Este expediente se encuentra en su estado inicial. No tiene movimientos registrados.</p>';
                return;
            }
            const h = data[0];
            contenedor.innerHTML = `
                <div class="text-xs text-gray-400 mb-1 font-semibold tracking-wide">⏳ Último movimiento</div>
                <table class="w-full text-left border-collapse text-[11px]">

                            <th class="p-2">Gerencia</th>

                            <th class="p-2">Receptor</th>
                            <th class="p-2">Estatus</th>
                            <th class="p-2 text-center">F. Recibido</th>
                            <th class="p-2 text-center">F. Devuelto</th>
                        </tr>
                    </thead>
                    <tbody class="divide-y divide-gray-800">
                        <tr class="hover:bg-gray-800/50 cursor-pointer text-teal-300/80" onclick="expandirHistorialCompleto(${id})">
                            <td class="p-2 font-mono font-bold">#1</td>

        function expandirHistorialCompleto(id) {
            const contenedor = document.getElementById(`hist-content-${id}`);
            const res = db.exec(`
                SELECT h.id_movimiento,
                    COALESCE(g.nombre,'N/A') AS ger, COALESCE(em.nombre,'N/A') AS emisor,
                    COALESCE(rec.nombre,'N/A') AS receptor, COALESCE(ed.nombre,'N/A') AS estatus,
                    COALESCE(h.fecha_recibido,'') AS f_entrada, COALESCE(h.fecha_devuelto,'') AS f_salida,
                    h.observaciones_generales
                FROM historial_movimientos h
                LEFT JOIN cat_gerencia g ON h.id_gerencia = g.id
                LEFT JOIN cat_responsables em ON h.id_emisor = em.id
                LEFT JOIN cat_responsables rec ON h.id_receptor = rec.id
                LEFT JOIN cat_estatus_detalle ed ON h.id_estatus = ed.id
                WHERE h.id_expediente = ${id}
                ORDER BY h.id_movimiento DESC
            `);
            const historial = dbToObjects(res);
            if (historial.length === 0) {
                contenedor.innerHTML = '<p class="text-gray-500 italic p-2">Sin movimientos registrados.</p>';
                return;
            }
            let html = `
                <div class="text-xs text-teal-400 mb-1 font-semibold tracking-wide">⏳ Historial completo (${historial.length} movimientos)</div>
                <table class="w-full text-left border-collapse text-[11px]">
                    <thead>
                        <tr class="text-gray-400 border-b border-gray-700/60 bg-gray-800/40 font-semibold">
                            <th class="p-2">Detalle</th>
                            <th class="p-2">Mov.</th>
                            <th class="p-2">Gerencia</th>
                            <th class="p-2">Emisor</th>
                            <th class="p-2">Receptor</th>
                            <th class="p-2">Estatus</th>
                            <th class="p-2 text-center">F. Recibido</th>
                            <th class="p-2 text-center">F. Devuelto</th>
                        </tr>
                    </thead>
                    <tbody class="divide-y divide-gray-800">
            `;
            historial.forEach((h, index) => {
                const mid = h.id_movimiento;
                const obs = h.observaciones_generales || '';
                html += `<tr class="hover:bg-gray-800/50 cursor-pointer" onclick="toggleDetalleMov(${mid}, event)">`;
                html += `<td class="p-2 text-center"><span class="text-amber-400" id="mbtn-${mid}">➕</span></td>`;
                html += `<td class="p-2 font-mono text-gray-500 font-bold">#${historial.length - index}</td>`;
                html += `<td class="p-2 text-teal-300 font-medium">${h.ger}</td>`;
                html += `<td class="p-2">${h.emisor}</td>`;
                html += `<td class="p-2">${h.receptor}</td>`;
                html += `<td class="p-2"><span class="px-1.5 py-0.5 rounded text-[10px] font-medium ${getEstatusClass(h.estatus)}">${h.estatus}</span></td>`;
                html += `<td class="p-2 text-center font-mono text-gray-300">${h.f_entrada}</td>`;
                html += `<td class="p-2 text-center font-mono text-gray-300">${h.f_salida}</td>`;
                html += `</tr>`;
                html += `<tr id="mdet-${mid}" class="hidden"><td colspan="8" style="padding:0;background:rgba(0,0,0,0.2)"><div style="padding:0.75rem 1rem;display:grid;grid-template-columns:repeat(auto-fill,minmax(180px,1fr));gap:0.5rem;font-size:0.75rem">`;
                html += `<div><span style="color:#6B7280;display:block;font-size:0.6875rem">ID Movimiento</span><span style="color:#D1D5DB;font-weight:500">#${mid}</span></div>`;
                html += `<div><span style="color:#6B7280;display:block;font-size:0.6875rem">Gerencia</span><span style="color:#D1D5DB;font-weight:500">${h.ger}</span></div>`;
                html += `<div><span style="color:#6B7280;display:block;font-size:0.6875rem">Emisor</span><span style="color:#D1D5DB;font-weight:500">${h.emisor}</span></div>`;
                html += `<div><span style="color:#6B7280;display:block;font-size:0.6875rem">Receptor</span><span style="color:#D1D5DB;font-weight:500">${h.receptor}</span></div>`;
                html += `<div><span style="color:#6B7280;display:block;font-size:0.6875rem">Estatus</span><span style="color:#D1D5DB;font-weight:500">${h.estatus}</span></div>`;
                html += `<div><span style="color:#6B7280;display:block;font-size:0.6875rem">F. Recibido</span><span style="color:#D1D5DB;font-weight:500">${h.f_entrada}</span></div>`;
                html += `<div><span style="color:#6B7280;display:block;font-size:0.6875rem">F. Devuelto</span><span style="color:#D1D5DB;font-weight:500">${h.f_salida}</span></div>`;
                if (obs) {
                    html += `<div style="grid-column:1/-1"><span style="color:#6B7280;display:block;font-size:0.6875rem">Observaciones</span><span style="color:#FBBF24;font-style:italic;background:rgba(0,0,0,0.3);padding:0.5rem;border-radius:0.375rem;display:block;margin-top:0.25rem">${obs}</span></div>`;
                }
                html += `</div></td></tr>`;
            });
            html += `</tbody></table>`;
            contenedor.innerHTML = html;
        }

        function expandirHistorialFormulario(id) {

            const res = db.exec(`
                SELECT h.id_movimiento,
                    COALESCE(g.nombre,'N/A') AS ger, COALESCE(em.nombre,'N/A') AS emisor,
                    COALESCE(rec.nombre,'N/A') AS receptor, COALESCE(ed.nombre,'N/A') AS estatus,

    OLD.id_estatus IS NOT NEW.id_estatus OR
    OLD.id_estado_accion IS NOT NEW.id_estado_accion OR
    OLD.fecha_recibido IS NOT NEW.fecha_recibido OR
    OLD.fecha_devuelto IS NOT NEW.fecha_devuelto OR
    OLD.observaciones_generales IS NOT NEW.observaciones_generales
)
BEGIN
    INSERT INTO historial_movimientos (
        id_expediente, id_tipo_contrato, id_gerencia, id_superintendencia, 
        id_documento, id_emisor, id_receptor, id_estatus, id_estado_accion,
        fecha_recibido, fecha_devuelto, observaciones_generales
    ) VALUES (
        NEW.id_expediente, NEW.id_tipo_contrato, NEW.id_gerencia, NEW.id_superintendencia, 
        NEW.id_documento, NEW.id_emisor, NEW.id_receptor, NEW.id_estatus, NEW.id_estado_accion,
        NEW.fecha_recibido, NEW.fecha_devuelto, NEW.observaciones_generales
    );

    UPDATE expedientes
    SET id_estatus = (SELECT id FROM cat_estatus_detalle WHERE nombre = 'PENDIENTE' LIMIT 1)
    WHERE NEW.fecha_firma_contrato IS NULL
      AND OLD.fecha_firma_contrato IS NOT NULL
      AND id_expediente = NEW.id_expediente;

    UPDATE expedientes
    SET fecha_actualizacion = CURRENT_DATE
    WHERE id_expediente = NEW.id_expediente;
END;

CREATE VIEW vw_historial_celdas_multilinea AS
SELECT 
    id_expediente,
    GROUP_CONCAT(nro_cambio || ': ' || gerencia || '   ', CAST(x'0A' AS TEXT) || '  |  ') AS hist_gerencia,
    GROUP_CONCAT(nro_cambio || ': ' || superintendencia || '   ', CAST(x'0A' AS TEXT) || '  |  ') AS hist_superintendencia,
    GROUP_CONCAT(nro_cambio || ': ' || emisor || '   ', CAST(x'0A' AS TEXT) || '  |  ') AS hist_emisor,
    GROUP_CONCAT(nro_cambio || ': ' || receptor || '   ', CAST(x'0A' AS TEXT) || '  |  ') AS hist_receptor,
    GROUP_CONCAT(nro_cambio || ': ' || estatus || '   ', CAST(x'0A' AS TEXT) || '  |  ') AS hist_estatus,
    GROUP_CONCAT(nro_cambio || ': ' || fecha_entrada || '   ', CAST(x'0A' AS TEXT) || '  |  ') AS hist_fechas_entrada,
    GROUP_CONCAT(nro_cambio || ': ' || fecha_salida || '   ', CAST(x'0A' AS TEXT) || '  |  ') AS hist_fechas_salida
FROM (
    SELECT 
        h.id_expediente,
        'Cambio ' || (
            SELECT COUNT(*) 
            FROM historial_movimientos h2 
            WHERE h2.id_expediente = h.id_expediente AND h2.id_movimiento <= h.id_movimiento
        ) AS nro_cambio,
        COALESCE(g.nombre, 'N/A') AS gerencia,
        COALESCE(s.nombre, 'N/A') AS superintendencia,
        COALESCE(em.nombre, 'N/A') AS emisor,
        COALESCE(rec.nombre, 'N/A') AS receptor,
        COALESCE(ed.nombre, 'N/A') AS estatus,
        COALESCE(h.fecha_recibido, 'Sin fecha') AS fecha_entrada,
        COALESCE(h.fecha_devuelto, 'Sin fecha') AS fecha_salida
    FROM historial_movimientos h
    LEFT JOIN cat_gerencia g ON h.id_gerencia = g.id
    LEFT JOIN cat_superintendencia s ON h.id_superintendencia = s.id
    LEFT JOIN cat_responsables em ON h.id_emisor = em.id
    LEFT JOIN cat_responsables rec ON h.id_receptor = rec.id
    LEFT JOIN cat_estatus_detalle ed ON h.id_estatus = ed.id
    ORDER BY h.id_expediente, h.id_movimiento DESC
)
GROUP BY id_expediente;

# Gestión de Expedientes con Historial — Documentación

## Contexto Termux (Android)

Este proyecto se edita y construye desde **Termux** en Android. Si inicias una sesión nueva:

| Ítem | Valor |
|------|-------|
| Directorio | `/storage/emulated/0/baseaccess` |
| Repositorio | `git@github.com:JulianoJul/baseaccess.git` |
| Node.js | `pkg install nodejs` (si no está) |
| Descargas | `curl` viene preinstalado |

**Comandos clave para reconstruir el `.exe`:**
```bash
npm install --save-dev --no-bin-links electron@latest electron-builder@latest
node node_modules/electron-builder/cli.js --win portable --x64
```
El build se genera en `dist/win-unpacked/`. Copiar esa carpeta a USB y ejecutar `GestionExpedientes.exe`.

**Importante:** `node_modules/` y `dist/` no se suben a git (`.gitignore`). Hay que reinstalar dependencias cada sesión nueva.

## ⚠️ Limitación: `file://` + WASM

Al abrir `index.html` con doble click (`file://` protocol), los navegadores **bloquean la carga del binario WASM** por seguridad. Síntomas:
- El botón "+ Nuevo Expediente" queda deshabilitado
- Los registros de la BDD no se muestran en la tabla

**Usar siempre el modo Electron portable** (`dist/win-unpacked/GestionExpedientes.exe`) para evitar este problema.

## Arquitectura

App web 100% cliente-side. **HTML + Tailwind CSS = UI** | **sql.js (SQLite WASM) = Data Layer**.
Sin backend, sin servidor, sin runtime externo. Un solo archivo HTML.

Dos modos de ejecución:

1. **Navegador** — abrir `index.html` directo (dependencias locales en `vendor/`)
2. **Electron portable** — `GestionExpedientes.exe` con Chromium embebido (sin depender de Firefox/Chrome)

## Electron Portable

Para no depender de ningún navegador, se puede construir un `.exe` portable:

### Source files
- `main.js` — Electron main process (ventana 1400x900, sin menú)
- `package.json` — `electron` + `electron-builder` como devDeps

### Build (requiere Node.js + npm)
```bash
npm install --save-dev --no-bin-links electron@latest electron-builder@latest
node node_modules/electron-builder/cli.js --win portable --x64
```

El `.exe` portable se genera en `dist/` (~80MB con Chromium embebido). Se ejecuta sin instalación, sin admin, sin depender del navegador del sistema.

> **Nota:** En Android/Termux el paso de empaquetado NSIS falla por falta de 7zip para ARM. Usar `--x64` para que el build se despliegue en `dist/win-unpacked/` — esa carpeta (~360MB) es funcional: copiar a USB, ejecutar `GestionExpedientes.exe` directo.

## Makefile

```bash
make combine          # Concatena index.html + Tablas7.sql + main.js + package.json + doc.md → combined.txt
make clean            # rm -f combined.txt
make commit msg="x"   # git add -A + git commit
make push             # git push
make github msg="x"   # commit + push (shortcut)
make serve            # python3 -m http.server 8000 (sirve index.html por HTTP para evitar file://)
```

## Reglas del Proceso

1. **doc.md primero**: antes de cualquier implementación o cambio de código, actualizar esta documentación con lo que se planea hacer.
2. **Makefile siempre**: después de cambios, ejecutar `make combine`.
3. **Sin hardcodeo**: cero assumptions de naming conventions. Toda heurística debe ser configurable.
4. **Historial de cambios**: cada cambio debe agregarse a la cronología en `doc.md` con fecha, archivo, y razón.
5. **DRY + Reutilización**: toda pieza de lógica debe tener una representación única. No repetir código ni copiar-pegar bloques. Si un patrón aparece en más de un lugar, extraer a función reutilizable. La modularidad no se mide en líneas por archivo ni por función, sino en ausencia de redundancia y en que cada función tenga una única responsabilidad (SRP). Una función de 200 líneas sin duplicación interna es mejor que 4 funciones de 50 líneas con lógica repetida.
